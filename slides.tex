\section{Phương pháp}
% ══════════════════════════════════════════════════════════════════════════════

\begin{frame}{Phương pháp: Apriori \& FP-Growth}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Thuật toán Apriori}
        \begin{itemize}
          \item Sinh tập phổ biến theo từng mức (breadth-first)
          \item Cắt tỉa các tập không đạt ngưỡng support
          \item Dễ hiểu, trực quan
          \item Chậm hơn khi dữ liệu lớn
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Thuật toán FP-Growth}
        \begin{itemize}
          \item Xây dựng cây FP-tree nén
          \item Không sinh ứng viên -- nhanh hơn nhiều
          \item Cho kết quả \textbf{giống hệt Apriori}
          \item Phù hợp dữ liệu lớn
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}

  \vspace{0.8em}

  \begin{block}{Tham số thực nghiệm}
    \centering
    \begin{tabular}{lll}
      \toprule
      \textbf{Tham số} & \textbf{Giá trị} & \textbf{Ý nghĩa} \\
      \midrule
      Min Support    & 0,5\% & Xuất hiện trong $\geq 565$ tweet \\
      Min Confidence & 10\%  & Luật đúng $\geq 10\%$ trường hợp \\
      Min Lift       & 1,5   & Mạnh hơn ngẫu nhiên $\geq 1,5$ lần \\
      \bottomrule
    \end{tabular}
  \end{block}
\end{frame}

\begin{frame}{Các độ đo đánh giá luật}
  \begin{block}{Định nghĩa}
    Cho luật $A \Rightarrow B$ trên tập giao dịch $T$:
  \end{block}
  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \[
        \mathrm{support}(A \Rightarrow B) = \frac{|A \cup B|}{|T|}
      \]
      \[
        \mathrm{confidence}(A \Rightarrow B) = \frac{|A \cup B|}{|A|}
      \]
    \end{column}
    \begin{column}{0.48\textwidth}
      \[
        \mathrm{lift}(A \Rightarrow B) = \frac{\mathrm{confidence}(A \Rightarrow B)}{\mathrm{support}(B)}
      \]
      \begin{itemize}
        \item $\mathrm{lift} > 1$: liên kết dương (tốt)
        \item $\mathrm{lift} = 1$: độc lập (không có nghĩa)
        \item $\mathrm{lift} < 1$: liên kết âm
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

      \textbf{Thuật toán}    & Apriori + FP-Growth (kết quả như nhau -- xác nhận độ chính xác) \\
      \textbf{Luật tìm được} & 27 luật có nghĩa \\
      \textbf{Phát hiện chính} & Cụm 4 hashtag tin tức, lift lên đến 143 \\
      \bottomrule
    \end{tabular}
  \end{block}

  \vspace{0.4em}

  \begin{exampleblock}{Ứng dụng thực tiễn}
    \begin{itemize}
      \item \textbf{Theo dõi thông tin dịch bệnh:} Người phát tán thông tin dùng bộ hashtag nhất quán -- có thể dùng để phát hiện tài khoản bot hoặc chiến dịch thông tin sai lệch.
      \item \textbf{Phân tích cộng đồng:} 5 cộng đồng hashtag phản ánh 5 góc nhìn khác nhau về đại dịch (y tế, chính trị, địa lý, hành vi, tổng quát).
      \item \textbf{Phân tích ảnh hưởng:} Một số hashtag được khuếch đại có chủ đích bởi tài khoản lớn.
    \end{itemize}
  \end{exampleblock}
\end{frame}
